# Local git init # Remote (cloning an existing repository) git clone # Add a specific file to the staging area git add # Add all files in the current directory git add . # Commit with a message git commit -m "Your commit message" # Show the working directory and staging area status git status # View the commit history git log # Create a new branch git branch # Switch to a branch git checkout # Create a new branch and switch to it git checkout -b # Merge another branch into the current branch git merge # Fetch and merge changes from the remote repository git pull origin # Push changes to the remote repository git push origin # Fetch changes from the remote repository without merging git fetch origin # Revert a commit by its hash (undoes the commit but keeps the changes) git revert # Reset the current branch to a specific commit (soft, mixed, or hard) git reset --soft git reset --mixed # default git reset --hard # Remove a file from the working directory and stage the removal git rm # Remove a file from Git but keep it locally git rm --cached
